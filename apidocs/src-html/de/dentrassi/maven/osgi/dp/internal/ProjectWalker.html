011package de.dentrassi.maven.osgi.dp.internal;
012
013import java.io.IOException;
014
015import org.apache.maven.artifact.Artifact;
016import org.apache.maven.project.MavenProject;
017
018public class ProjectWalker implements ArtifactWalker {
019
020    private final MavenProject project;
021
022    public ProjectWalker(final MavenProject project) {
023        this.project = project;
024    }
025
026    @Override
027    public void walk(final ArtifactConsumer consumer) {
028        final Artifact art = this.project.getArtifact();
029
030        if (art.getFile() == null) {
031            return;
032        }
033
034        if (art.getFile().getName().toLowerCase().endsWith(".jar")) {
035            try {
036                consumer.accept(new Entry(art.getFile().toPath()));
037            } catch (final IOException e) {
038                throw new RuntimeException(e);
039            }
040        }
041    }
042
043}
