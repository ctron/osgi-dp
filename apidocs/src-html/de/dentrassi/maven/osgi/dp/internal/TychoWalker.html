011package de.dentrassi.maven.osgi.dp.internal;
012
013import static java.lang.String.format;
014
015import java.io.File;
016import java.io.IOException;
017
018import org.apache.maven.plugin.logging.Log;
019import org.eclipse.tycho.ReactorProject;
020import org.eclipse.tycho.core.ArtifactDependencyVisitor;
021import org.eclipse.tycho.core.ArtifactDependencyWalker;
022import org.eclipse.tycho.core.FeatureDescription;
023import org.eclipse.tycho.core.PluginDescription;
024
025public class TychoWalker implements ArtifactWalker {
026
027    private final ArtifactDependencyWalker walker;
028    private final Log logger;
029
030    public TychoWalker(final ArtifactDependencyWalker walker, final Log logger) {
031        this.walker = walker;
032        this.logger = logger;
033    }
034
035    @Override
036    public void walk(final ArtifactConsumer consumer) {
037        this.walker.walk(new ArtifactDependencyVisitor() {
038            @Override
039            public boolean visitFeature(final FeatureDescription feature) {
040                return true;
041            }
042
043            @Override
044            public void visitPlugin(final PluginDescription plugin) {
045                try {
046                    TychoWalker.this.logger.debug(format("Considering artifact: %s", plugin));
047
048                    if (!"eclipse-plugin".equalsIgnoreCase(plugin.getKey().getType())) {
049                        TychoWalker.this.logger.debug(format("Not a JAR file -> %s", plugin.getKey().getType()));
050                        return;
051                    }
052
053                    final File location;
054                    final ReactorProject p = plugin.getMavenProject();
055                    if (p != null) {
056                        location = p.getArtifact();
057                    } else {
058                        location = plugin.getLocation();
059                    }
060
061                    if (location == null) {
062                        TychoWalker.this.logger.warn(format("Unable to locate artifact: %s", plugin));
063                    }
064
065                    if (!location.isFile()) {
066                        TychoWalker.this.logger.warn(format("Location '%s' is not a file", location));
067                        return;
068                    }
069
070                    consumer.accept(new Entry(location.toPath()));
071                } catch (final IOException e) {
072                    throw new RuntimeException(e);
073                }
074            }
075        });
076    }
077
078}
