069    private static final DateFormat TIMESTAMP_FORMAT = new SimpleDateFormat("yyyyMMddHHmmss");

141    public AbstractDpMojo() {

145    protected ArtifactWalker lookupDependencyWalker() {
146        final String packaging = this.project.getPackaging();
147
148        if (!this.tychoWalkerProjects.contains(packaging)) {
149            return new ProjectWalker(this.project);
150        }
151
152        final TychoProject facet = this.projectTypes.get(packaging);
153        if (facet == null) {
154            throw new IllegalStateException(format("Unknown packaging '%s'", packaging));
155        }
156
157        return new TychoWalker(facet.getDependencyWalker(this.project), getLog());
158    }
159
160    public void setVersion(final String version) {
161        this.version = Version.parseVersion(version).toString();
162    }
163
164    @Override
165    public void execute() throws MojoExecutionException, MojoFailureException {
166        if (this.skip) {
167            return;
168        }
169
170        final String dpVersion = makeVersion(true);
171
172        getLog().info("Building DP - Version: " + dpVersion);
173
174        final Manifest dpmf = new Manifest();
175        dpmf.getMainAttributes().put(Attributes.Name.MANIFEST_VERSION, "1");
176        dpmf.getMainAttributes().putValue("DeploymentPackage-SymbolicName", this.project.getArtifactId());
177        dpmf.getMainAttributes().putValue("DeploymentPackage-Version", dpVersion);
178
179        try {
180            final Map<String, File> files = new HashMap<>();
181
182            fillFromDependencies(dpmf, files);
183
184            final String dpName = String.format("%s_%s.dp", this.project.getArtifactId(), makeVersion(false));
185            final Path out = Paths.get(this.project.getBuild().getDirectory(), dpName);
186
187            getLog().info("Writing to: " + out);
188
189            Files.createDirectories(out.getParent());
190
191            try (JarOutputStream jar = new JarOutputStream(Files.newOutputStream(out), dpmf)) {
192
193                for (final Map.Entry<String, File> entry : files.entrySet()) {
194
195                    final Path p = entry.getValue().toPath();
196                    final JarEntry je = new JarEntry(entry.getKey());
197
198                    je.setSize(Files.size(p));
199                    je.setLastModifiedTime(Files.getLastModifiedTime(p));
200
201                    jar.putNextEntry(je);
202                    try (InputStream in = Files.newInputStream(p)) {
203                        ByteStreams.copy(in, jar);
204                    }
205                }
206            }
207
208            attach(out);
209
210        } catch (final IOException e) {
211            throw new MojoFailureException("Failed to process", e);
212        }
213    }
214
215    protected abstract void attach(Path out);
216
217    protected void fillFromDependencies(final Manifest manifest, final Map<String, File> files)
218            throws IOException, MojoExecutionException {
219
220        // from dependency walker
221
222        fillFromDependencyWalker(manifest, files);
223
224        // additional dependencies
225
226        fillfromExtraDependencies(manifest, files);
227    }
228
229    protected void fillfromExtraDependencies(final Manifest manifest, final Map<String, File> files)
230            throws IOException, MojoExecutionException {
231
232        if (this.additionalDependencies == null) {
233            // no additional dependencies
234            return;
235        }
236
237        try {
238            final Collection<ArtifactRequest> requests = new ArrayList<>(this.additionalDependencies.length);
239
240            for (final Dependency dep : this.additionalDependencies) {
241                final DefaultArtifact art = new DefaultArtifact(dep.getGroupId(), dep.getArtifactId(),
242                        dep.getClassifier(), dep.getType(), dep.getVersion());
243                final org.eclipse.aether.graph.Dependency adep = new org.eclipse.aether.graph.Dependency(art,
244                        JavaScopes.RUNTIME);
245                requests.add(new ArtifactRequest(new DefaultDependencyNode(adep)));
246            }
247
248            final List<ArtifactResult> result = this.resolver.resolveArtifacts(this.repositorySession, requests);
249
250            for (final ArtifactResult ares : result) {
251                getLog().debug("Additional dependency: " + ares);
252                processArtifact(manifest, files, ares.getArtifact().getFile());
253            }
254        } catch (final ArtifactResolutionException e) {
255            throw new MojoExecutionException("Failed to resolve additional dependencies", e);
256        }
257    }
258
259    protected void fillFromDependencyWalker(final Manifest manifest, final Map<String, File> files) {
260        final ArtifactWalker dw = lookupDependencyWalker();
261        dw.walk(entry -> processArtifact(manifest, files, entry.getLocation().toFile()));
262    }
263
264    protected void processArtifact(final Manifest dpmf, final Map<String, File> files, final File location)
265            throws IOException {
266
267        try (JarFile jar = new JarFile(location)) {
268            final Manifest mf = jar.getManifest();
269            if (mf == null) {
270                getLog().debug("No Manifest");
271                return;
272            }
273
274            String bsn = mf.getMainAttributes().getValue("Bundle-SymbolicName");
275            if (bsn == null) {
276                getLog().debug("No BSN");
277                return;
278            }
279            if (bsn != null) {
280                bsn = bsn.split(";", 2)[0];
281            }
282
283            final String version = mf.getMainAttributes().getValue("Bundle-Version");
284
285            final String fn = String.format("%s_%s.jar", bsn, version);
286
287            final Attributes attrs = new Attributes();
288            attrs.putValue("Bundle-SymbolicName", bsn);
289            attrs.putValue("Bundle-Version", version);
290            dpmf.getEntries().put(fn, attrs);
291
292            getLog().info(format("Added: %s:%s", bsn, version));
293
294            files.put(fn, location);
295        }
296    }
297
298    private String makeVersion(final boolean osgiVersion) {
299        if (this.version != null) {
300            return this.version.toString();
301        }
302
303        if (this.useQualifiedFilename) {
304            return makeQualifiedVersion().toString();
305        }
306
307        if (osgiVersion) {
308            return makeQualifiedVersion().toString();
309        } else {
310            return this.project.getVersion();
311        }
312    }
313
314    private Version makeQualifiedVersion() {
315        try {
316            final ReactorProject rp = (ReactorProject) this.project.getContextValue(ReactorProject.CTX_REACTOR_PROJECT);
317            if (rp != null) {
318                return new Version(rp.getExpandedVersion());
319            }
320        } catch (final Exception e) {
321            getLog().debug("Failed to get qualified tycho version", e);
322        }
323
324        String version = this.project.getVersion();
325        if (version.endsWith("-SNAPSHOT")) {
326            version = version.replaceAll("-SNAPSHOT$", "." + TIMESTAMP_FORMAT.format(this.session.getStartTime()));
327        }
328
329        return new Version(version);
330    }
