011package de.dentrassi.maven.osgi.dp;
012
013import java.io.File;
014import java.io.IOException;
015import java.nio.file.Path;
016import java.util.HashSet;
017import java.util.Map;
018import java.util.Set;
019import java.util.jar.Manifest;
020
021import org.apache.maven.artifact.Artifact;
022import org.apache.maven.plugin.MojoExecutionException;
023import org.apache.maven.plugins.annotations.LifecyclePhase;
024import org.apache.maven.plugins.annotations.Mojo;
025import org.apache.maven.plugins.annotations.Parameter;
026import org.apache.maven.plugins.annotations.ResolutionScope;
027
028/**
029 * Build this project as an OSGi distribution package.
030 * <p>
031 * This project takes the main attachment and the dependencies of this project
032 * and builds an OSGi DP from it. By default only direct {@code compile} and
033 * {@code runtime} dependencies will be added to the package. All extra
034 * dependencies will be included.
035 * </p>
036 *
037 * @author Jens Reimann
038 * @since 0.4.0
039 */
040@Mojo(name = "bundle", defaultPhase = LifecyclePhase.PACKAGE, requiresProject = true, requiresDependencyResolution = ResolutionScope.RUNTIME, requiresDependencyCollection = ResolutionScope.RUNTIME, threadSafe = false)
041public class BundleMojo extends AbstractDpMojo {
042
043    /**
044     * Whether or not the resulting DP should be attached to the project output.
045     */
046    @Parameter(defaultValue = "true")
047    protected boolean attach = true;
048
049    /**
050     * Whether or not to fail if the main artifact is missing
051     */
052    @Parameter(defaultValue = "true")
053    protected boolean failOnMissingMainArtifact = true;
054
055    /**
056     * Allows to ignore the project dependencies and only use the extra
057     * dependencies.
058     */
059    @Parameter(defaultValue = "false")
060    protected boolean ignoreProjectDependecies = false;
061
062    /**
063     * The set of scopes for dependencies which will be included in the DP
064     */
065    @Parameter(defaultValue = "compile,runtime")
066    protected Set<String> includedScopes = new HashSet<>();
067
068    @Override
069    protected void attach(final Path out) {
070        if (!this.attach) {
071            return;
072        }
073
074        this.projectHelper.attachArtifact(this.project, "dp", out.toFile());
075    }
076
077    @Override
078    protected void fillFromDependencies(final Manifest manifest, final Map<String, File> files)
079            throws IOException, MojoExecutionException {
080
081        super.fillFromDependencies(manifest, files);
082
083        if (!this.ignoreProjectDependecies) {
084            for (final Artifact art : this.project.getDependencyArtifacts()) {
085
086                final String scope = art.getScope();
087                if (this.includedScopes.contains(scope)) {
088                    processArtifact(manifest, files, art.getFile());
089                }
090            }
091        }
092
093        processMainArtifact(manifest, files);
094    }
095
096    private void processMainArtifact(final Manifest manifest, final Map<String, File> files)
097            throws MojoExecutionException, IOException {
098
099        final File artifact = this.project.getArtifact().getFile();
100
101        if (artifact == null || !artifact.isFile()) {
102            if (this.failOnMissingMainArtifact) {
103                throw new MojoExecutionException(
104                        "The main artifact is missing. You may disable this error by setting 'failOnMissingMainArtifact' to 'true'");
105            }
106        }
107
108        if (artifact != null && artifact.isFile()) {
109            processArtifact(manifest, files, artifact);
110        }
111    }
112}
